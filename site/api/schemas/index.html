    def omop_columns(
    table: str,
    field: str | None = None,
    version: str = "5.3",
) -> tuple[str, ...] | str:
    """
    Return column names for an OMOP CDM table (or a single field mapping).

    Parameters
    ----------
    table : str
        CDM table name (e.g. "person", "observation_period") or cohort table.
    field : str, optional
        If set, return the column name for that logical field (e.g. "start_date", "end_date",
        "person_id") from FIELD_TABLES_COLUMNS. One of start_date, end_date, standard_concept,
        source_concept, type_concept, unique_id, domain_id, person_id.
    version : "5.3" | "5.4"
        CDM version.

    Returns
    -------
    If field is None: tuple of column names for the table.
    If field is set: the column name string for that field in the table (or KeyError if not mapped).
    """
    if version not in ("5.3", "5.4"):
        raise ValueError(f"Unsupported CDM version: {version}. Use '5.3' or '5.4'.")
    table_lower = table.lower()
    if field is not None:
        if table_lower not in FIELD_TABLES_COLUMNS:
            raise ValueError(
                f"Table {table!r} has no field mapping. "
                "Use omop_columns(table) without 'field' for column list, or a table in FIELD_TABLES_COLUMNS."
            )
        cols = FIELD_TABLES_COLUMNS[table_lower]
        if field not in cols:
            raise ValueError(
                f"Unknown field {field!r}. Choose from: {list(cols.keys())}"
            )
        val = cols[field]
        if val is None:
            raise ValueError(f"Table {table!r} has no column for field {field!r}.")
        return val
    # Return all columns for this table from omop_table_fields
    df = omop_table_fields(version)
    subset = df[df["cdm_table_name"] == table_lower]
    if len(subset) == 0:
        if table_lower in ("cohort", "cohort_set", "cohort_attrition"):
            return cohort_columns(table_lower)
        raise ValueError(
            f"Unknown table {table!r}. Use omop_tables() for CDM table names, "
            "or 'cohort', 'cohort_set', 'cohort_attrition' for cohort tables."
        )
    return tuple(subset["cdm_field_name"].tolist())

    def omop_table_fields(version: str = "5.3") -> Any:
    """
    Return OMOP table/field metadata as a DataFrame (cdm_table_name, cdm_field_name, is_required, cdm_datatype, type).

    Parameters
    ----------
    version : "5.3" | "5.4"
        CDM version. 5.4 uses the same core field set as 5.3 in this implementation.

    Returns
    -------
    pandas.DataFrame with columns cdm_table_name, cdm_field_name, is_required, cdm_datatype, type
    (type is one of "cdm_table", "cohort", "achilles").
    """
    import pandas as pd

    if version not in ("5.3", "5.4"):
        raise ValueError(f"Unsupported CDM version: {version}. Use '5.3' or '5.4'.")
    rows = _OMOP_FIELD_LEVEL_5_3 + _COHORT_FIELDS + _ACHILLES_FIELDS
    return pd.DataFrame(
        rows,
        columns=["cdm_table_name", "cdm_field_name", "is_required", "cdm_datatype", "type"],
    )

    def cohort_columns(table: str) -> tuple[str, ...]:
    """Return required columns for cohort table, cohort_set, or cohort_attrition."""
    if table == "cohort":
        return COHORT_TABLE_COLUMNS
    if table == "cohort_set":
        return COHORT_SET_COLUMNS
    if table == "cohort_attrition":
        return COHORT_ATTRITION_COLUMNS
    raise ValueError(f"Unknown cohort table: {table}. Use 'cohort', 'cohort_set', or 'cohort_attrition'.")
